import bpy


from vox_exporter.operators.base_operator_exporter import BaseOperatorExporter


class EXPORT_OT_{{code_name}}(BaseOperatorExporter):
    bl_idname = "export.{{code_name}}"
    bl_label = "{{name}} (.{{type}})"
    bl_description = "Export selected objects to {{name}} format (.{{type}})"
    bl_options = {'REGISTER', 'UNDO'}

    filename_ext = ".{{type}}"

    filter_glob: bpy.props.StringProperty(
        default="*.{{type}}",
        options={'HIDDEN'},
        maxlen=255,
    )


def on_file_export_{{type}}_click(self, context):
    self.layout.operator_context = 'INVOKE_DEFAULT'
    self.layout.operator(EXPORT_OT_{{code_name}}.bl_idname, text="{{name}} (.{{type}})")

def register():
    bpy.utils.register_class(EXPORT_OT_{{code_name}})
    bpy.types.TOPBAR_MT_file_export.append(on_file_export_{{type}}_click)

def unregister():
    bpy.utils.unregister_class(EXPORT_OT_{{code_name}})
    bpy.types.TOPBAR_MT_file_export.remove(on_file_export_{{type}}_click)
